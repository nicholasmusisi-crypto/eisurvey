function doPost(e) { try { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); // Upload audio to Drive and get a shareable link let audioUrl = ''; if (d.audioB64) { try { const folder = getOrCreateFolder_('E!Survey Audio'); const ext = d.audioExt || 'webm'; const bytes = Utilities.base64Decode(d.audioB64); const file = folder.createFile( Utilities.newBlob(bytes, 'audio/' + ext, (d.id || Date.now()) + '.' + ext) ); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); audioUrl = file.getUrl(); } catch(ae) { /* audio upload failed — row saved without link */ } } const nm = String(d.formTitle||'Entries').slice(0,100); let sh = ss.getSheetByName(nm)||ss.insertSheet(nm); const ans = d.answers||{}; const meta = ['_id','_createdAt','_completedAt','_enumerator','_status','_form','_audioSec','_audioLink','_quality']; let hdrs; if (sh.getLastRow()===0) { hdrs = [...meta,...Object.keys(ans)]; sh.appendRow(hdrs); sh.getRange(1,1,1,hdrs.length).setFontWeight('bold'); } else { hdrs = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0]; const nk = Object.keys(ans).filter(k=>!hdrs.includes(k)); if (nk.length) { hdrs=[...hdrs,...nk]; sh.getRange(1,1,1,hdrs.length).setValues([hdrs]); } } sh.appendRow(hdrs.map(h=>{ if(h==='_id') return String(d.id||''); if(h==='_createdAt') return d.createdAt||''; if(h==='_completedAt') return d.completedAt||''; if(h==='_enumerator') return d.enumerator||''; if(h==='_status') return d.status||''; if(h==='_form') return d.formTitle||''; if(h==='_audioSec') return d.durationSec||0; if(h==='_audioLink') return audioUrl; if(h==='_quality') return d.quality||''; return ans[h]!==undefined ? String(ans[h]) : ''; })); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet() { return ContentService.createTextOutput(JSON.stringify({status:'ok',msg:'E!Survey active'})) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateFolder_(name) { const it = DriveApp.getFoldersByName(name); return it.hasNext() ? it.next() : DriveApp.createFolder(name); }
